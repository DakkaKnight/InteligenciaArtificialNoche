from logic import *

# Definimos los símbolos
rain = Symbol("rain")               # Lluvia
bbc_visited = Symbol("bbc_visited") # Estudiantes visitaron BBC
unimayor_visited = Symbol("unimayor_visited") # Estudiantes visitaron Unimayor

# Definimos la base de conocimiento
knowledge = And(
    Implication(Not(rain), bbc_visited),  # Si no llueve, los estudiantes visitan a BBC hoy
    Or(bbc_visited, unimayor_visited),     # Estudiantes visitaron BBC o Unimayor, pero no ambos
    Not(And(bbc_visited, unimayor_visited)), # No visitaron ambos lugares
    unimayor_visited                       # Estudiantes visitaron Unimayor hoy
)

# Imprimimos el resultado de la comprobación para la lluvia
print("¿Está lloviendo?",model_check(knowledge, rain))  # Verifica si la proposición lluvia es verdadera
